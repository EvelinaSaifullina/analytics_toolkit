# 1. Инициализируй Git (если ещё не делала)
git init

# 2. Добавь все файлы проекта
git add .

# 3. Создай коммит
git commit -m "feat: add flight delay analysis with causes, airlines and routes"

# 4. Свяжи локальный репозиторий с удалённым
git remote add origin https://github.com/EvelinaSaifullina/flight_delay_analysis.git

# 5. Отправь изменения в GitHub
git push -u origin main

# Проверь, какие файлы будут отправлены
git status

# Если Git ругается на main vs master
git branch -M main
git push -u origin main
